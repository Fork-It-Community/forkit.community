import Prose from "@/components/Prose/index.astro";
import { getEntry, render } from "astro:content";
import { lunalink } from '@bearstudio/lunalink';
import { ROUTES } from "@/routes.gen";
import { Image } from "astro:assets";

export default async function CustomBlock() {
    const speaker = await getEntry("people", "erwan-rougeux");

    const afternoonKeynoteTalkForHanoi2026 = await getEntry(
        "talks",
        "digital-sovereignty-and-ai-the-strategic-cost-of-inaction",
    );

    const ContentKeynote = (await render(afternoonKeynoteTalkForHanoi2026)).Content

    return (
        <div class="relative mx-auto flex w-full max-w-screen-lg flex-col gap-12 px-4 pb-16 md:pb-24">
            <article class="flex flex-col gap-8 md:flex-row md:items-start">
                <div class="grid grid-cols-2 gap-4 md:flex-[0.5] md:grid-cols-1">
                    {speaker?.data.avatar && (
                        <div class="relative overflow-hidden rounded-xl">
                            <a
                                href={lunalink(ROUTES.people[":id"].__path, {
                                    id: speaker.id,
                                })}
                            >
                                <Image
                                    class="aspect-[3/4] w-full object-cover transition duration-500 hover:scale-105"
                                    src={speaker.data.avatar}
                                    alt={speaker.data.name}
                                    width={512}
                                    height={680}
                                />
                            </a>
                            <div class="absolute bottom-0 left-0 w-full bg-gradient-to-t from-black/80 to-transparent p-4 text-center">
                                <span class="text-sm font-bold text-white">
                                    {speaker.data.name}
                                </span>
                            </div>
                        </div>
                    )}
                </div>
                <div class="flex flex-col justify-center gap-6 md:flex-[1.5]">
                    <div class="flex flex-col gap-2">
                        <span class="text-sm font-bold uppercase tracking-widest text-primary">
                            Afternoon Keynote
                        </span>
                        <h3 class="font-heading text-2xl font-bold leading-tight md:text-4xl">
                            {afternoonKeynoteTalkForHanoi2026?.data.title}
                        </h3>
                    </div>
                    <Prose>{ContentKeynote && <ContentKeynote />}</Prose>
                </div>
            </article>
        </div>
    )
}
